20 {
21  int j;
22  float b0 = coeffs[0];
23  float a1 = coeffs[1];

26 
27  out[0] = b0*in[0] - a1*i0[0] - a2*i0[1] - a3*i0[2];
28  out[1*stepout] = b0*in[stepin] - a1*out[0] - a2*i0[0] - a3*i0[1];
29  out[2*stepout] = b0*in[2*stepin] - a1*out[stepout] - a2*out[0] - a3*i0[0];
